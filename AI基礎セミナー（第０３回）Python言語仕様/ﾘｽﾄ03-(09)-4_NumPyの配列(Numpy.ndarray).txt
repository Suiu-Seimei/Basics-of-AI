import numpy as np

#(1)  1次元の配列
npArray1 = np.array(["00","01","02"])
print("1次元の配列：\n{0}\n".format(npArray1))

#(2)  2次元の配列
npArray2 = np.array([["00","01","02"],["10","11","12"]])
print("2次元の配列：\n{0}\n".format(npArray2))
