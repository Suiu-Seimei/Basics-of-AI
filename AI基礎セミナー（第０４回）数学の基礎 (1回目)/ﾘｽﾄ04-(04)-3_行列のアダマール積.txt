import numpy as np

print("1×3型行列 A と B のｱﾀﾞﾏｰﾙ積")
A = np.array([1,2,3])
B = np.array([4,5,6])
print('Matrix A',A)
print('Matrix B',B)
print('A*B =', A*B)

print("\n2×3型行列 A と B のｱﾀﾞﾏｰﾙ積")
A = np.array([[11,12,13],[21,22,23]])
B = np.array([[1,2,3],[4,5,6]])
print('Matrix A', A)
print('Matrix B', B)
print('A*B =', A*B)
